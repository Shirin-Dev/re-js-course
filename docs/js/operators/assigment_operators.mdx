```jsx
var x = 1; // присвоить переменной "x" число 1
x += 3; // эквивалентно x = x + 3, сейчас x равен 4
x *= 3; // эквивалентно x = x * 3, сейчас x равен 12
x /= 2; // эквивалентно x = x / 2, сейчас x равен 6
x -= 5; // эквивалентно x = x - 5, сейчас x равен 1
console.log(x); // выведет в консоль 1
```
